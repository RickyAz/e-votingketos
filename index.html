0${paslon.id}

${paslon.nama}

Visi & Misi

${paslon.visi_misi}

`; }); document.getElementById('container-kandidat').innerHTML = htmlCard; document.getElementById('box-identitas').classList.add('hidden'); document.getElementById('box-bilik-suara').classList.remove('hidden'); } }) .catch(err => { alert(`Gagal Memuat Data.\n\nSolusi: Pastikan URL Apps Script Anda berakhiran /exec dan diset ke 'Anyone'.\n\nDetail: ${err.message}`); }) .finally(() => { btn.innerText = "Masuk ke Bilik Suara"; btn.disabled = false; }); } function eksekusiVoting(idPaslon) { if(!confirm(`Apakah Anda yakin ingin memilih Paslon 0${idPaslon}?`)) return; // Menggunakan text/plain untuk melewati blokir CORS Vercel secara total fetch(GAS_VOTING_URL, { method: "POST", mode: "cors", headers: { "Content-Type": "text/plain;charset=utf-8" }, body: JSON.stringify({ nis: nisPemilih, pilihan_id: idPaslon }) }) .then(res => res.json()) .then(res => { if(res.success) { alert("🎉 Terima kasih! Suara Anda telah berhasil direkam."); window.location.reload(); } else { alert(`🚫 Akses Ditolak:\n${res.error}`); window.location.reload(); } }) .catch(err => { alert("Gagal mengirim suara: " + err.message); }); }
